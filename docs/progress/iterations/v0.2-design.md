### 4.1 Source 类型自动识别 — `app/source_detector.py`

新建独立模块，纯函数无副作用，API 和 Worker 均可导入：

```python
from app.schemas import SourceType

# URL 域名/路径匹配表：(host_suffix, path_prefix, type)
# host_suffix 匹配规则：精确匹配 或 ".host_suffix" 结尾（防止子域名误判）
# path_prefix 为空时仅匹配域名
_DOMAIN_RULES: list[tuple[str, str, SourceType]] = [
    ("x.com", "", SourceType.x_twitter),
    ("twitter.com", "", SourceType.x_twitter),
    ("github.com", "", SourceType.github_trending),
    ("huggingface.co", "/papers", SourceType.hf_daily_papers),
    ("news.ycombinator.com", "", SourceType.hacker_news),
    ("semanticscholar.org", "", SourceType.semantic_scholar),
]

# URL 路径模式（用于 RSS 识别）
_RSS_PATH_PATTERNS = ["/feed/", "/rss/", "/feeds/posts/default"]
_RSS_SUFFIXES = [".xml", ".rss", ".atom"]
_RSS_CONTENT_TYPES = ["application/rss+xml", "application/atom+xml"]


def _host_matches(host: str, suffix: str) -> bool:
    """域名精确匹配或子域名匹配（避免 x.com 误匹配 x.com.evil.com）。"""
    return host == suffix or host.endswith("." + suffix)


def detect_type_from_url(source_url: str) -> SourceType:
    """基于 URL 特征识别 Source 类型（不含 HTTP 请求）。"""
    from urllib.parse import urlparse
    parsed = urlparse(source_url)
    host = (parsed.hostname or "").lower()
    path = parsed.path.lower()

    for host_suffix, path_prefix, st in _DOMAIN_RULES:
        if _host_matches(host, host_suffix):
            if not path_prefix or path.startswith(path_prefix):
                return st

    for pat in _RSS_PATH_PATTERNS:
        if pat in path:
            return SourceType.rss

    for suf in _RSS_SUFFIXES:
        if path.endswith(suf):
            return SourceType.rss

    return SourceType.unknown


async def detect_type_with_http(source_url: str, timeout: float = 3.0) -> SourceType:
    """当 URL 特征无法判定时，发起 HEAD 请求检查 Content-Type。"""
    base = detect_type_from_url(source_url)
    if base != SourceType.unknown:
        return base

    import httpx
    from app.settings import settings
    proxy = settings.https_proxy or settings.http_proxy
    try:
        async with httpx.AsyncClient(timeout=timeout, proxy=proxy, follow_redirects=True) as client:
            resp = await client.head(source_url)
            ct = resp.headers.get("content-type", "")
            for rss_ct in _RSS_CONTENT_TYPES:
                if rss_ct in ct:
                    return SourceType.rss
    except Exception:
        pass

    return SourceType.unknown
```

`detect_type_from_url` 在创建 Source 时同步调用（不阻塞）；仅当结果 `unknown` 时触发 `detect_type_with_http`（可能有 3s 延迟）。

Source 状态机中 `verified → active` 的转换不由用户手动触发，而是在 Source 首次被启用参与正式抓取时自动完成。

### 4.3 Source 验证与标记

#### POST /v1/sources/{source_id}/verify — 测试抓取

```python
@router.post("/v1/sources/{source_id}/verify", response_model=SourceVerifyResponse)
async def verify_source(source_id: uuid.UUID, session: AsyncSession = Depends(get_session)):
    row = (await session.execute(select(Source).where(Source.id == source_id))).scalar_one_or_none()
    if not row:
        raise HTTPException(status_code=404, detail="source not found")

    source_type = row.type
    config = _as_dict(row.config)

    # 根据 Source type 调用对应的抓取函数
    try:
        items = await asyncio.wait_for(
            _execute_verify_fetch(source_type, config),
            timeout=15.0,
        )
    except asyncio.TimeoutError:
        await session.execute(
            update(Source).where(Source.id == source_id).values(verify_error="外部服务超时（15s）")
        )
        await session.commit()
        return SourceVerifyResponse(status="error", error="外部服务超时（15s）")

    except NonRetryableError as e:
        await session.execute(
            update(Source).where(Source.id == source_id).values(verify_error=str(e))
        )
        await session.commit()
        return SourceVerifyResponse(status="error", error=str(e))

    except Exception as e:
        await session.execute(
            update(Source).where(Source.id == source_id).values(verify_error=str(e))
        )
        await session.commit()
        return SourceVerifyResponse(status="error", error=str(e))

    # 验证成功 — 映射 parse_x_twitter 返回格式到 VerifyItem
    # parse_x_twitter 返回的 item 结构：{source_item_id, url, published_at, content:{text, author_name, ...}}
    verify_items = [
        VerifyItem(
            source_item_id=it["source_item_id"],
            source_item_url=it.get("url"),
            title=it.get("content", {}).get("text", "")[:80] if isinstance(it.get("content"), dict) else "",
            content_preview=_truncate_preview(it, 200),
            published_at=str(it.get("published_at")) if it.get("published_at") else None,
        )
        for it in items[:5]
    ]

    await session.execute(
        update(Source).where(Source.id == source_id).values(
            last_verified_at=datetime.now(tz=UTC),
            verify_error=None,
        )
    )
    await session.commit()
    return SourceVerifyResponse(status="ok", items=verify_items, total_fetched=len(items))
```

`_execute_verify_fetch` 是分发函数：

```python
async def _execute_verify_fetch(source_type: str, config: dict) -> list[dict]:
    """验证抓取分发——仅调用抓取函数，不入库。"""
    if source_type == "x_twitter":
        from worker.fetch_x_twitter import parse_x_twitter
        items, _ = await parse_x_twitter(config=config, cursor={}, max_items=5)
        return items
    # 其他类型暂未实现抓取器
    raise NonRetryableError(f"该 Source 类型的抓取器尚未实现：{source_type}")
```

**关键设计决策**：API 直接导入 `worker/fetch_x_twitter.parse_x_twitter`。该函数本身就是纯抓取逻辑（HTTP 调用 + 解析），不依赖 Worker 的数据库连接，从 API 调用是安全的。后续迭代新增抓取器时，只需在 `_execute_verify_fetch` 中新增分支。

`_truncate_preview` 辅助函数——从 parse_x_twitter item 提取文本预览：

```python
def _truncate_preview(item: dict, max_chars: int) -> str:
    """从 item 提取文本内容并截断。item 结构见 parse_x_twitter 返回格式。"""
    content = item.get("content", {})
    if isinstance(content, dict):
        text = content.get("text") or content.get("description") or ""
    else:
        text = str(content) if content else ""
    if len(text) <= max_chars:
        return text
    return text[:max_chars]
```

#### POST /v1/sources/{source_id}/mark-verified — 标记已验证

```python
@router.post("/v1/sources/{source_id}/mark-verified")
async def mark_source_verified(source_id: uuid.UUID, session: AsyncSession = Depends(get_session)):
    row = (await session.execute(select(Source).where(Source.id == source_id))).scalar_one_or_none()
    if not row:
        raise HTTPException(status_code=404, detail="source not found")

    if row.status in ("verified", "active"):
        return {"status": "ok"}  # 幂等

    if row.status == "error":
        raise HTTPException(status_code=409, detail="不能将 error 状态的 Source 直接标记为 verified")

    if row.status != "unverified":
        raise HTTPException(status_code=409, detail=f"当前状态 {row.status} 不允许此操作")

    await session.execute(
        update(Source).where(Source.id == source_id).values(status="verified")
    )
    await session.commit()
    return {"status": "ok"}
```

```python
@router.get("/channel-spaces/{channel_space_id}/news", response_model=list[ProcessedNewsOut])
async def list_news(
    channel_space_id: uuid.UUID,
    limit: int = Query(default=20, ge=1, le=100),
    offset: int = Query(default=0, ge=0),
    sub_channel_id: Optional[uuid.UUID] = Query(default=None),

    elif sort == "score_asc":

        .order_by(order_col, ProcessedNews.created_at.desc())
```

### 4.6 日志查询端点

#### GET /v1/admin/logs

文件读取策略：
- 源文件：`logs/api.log`、`logs/worker.log`（通过 `source` 参数区分，默认同时读取两者并合并排序）
- 读取方式：从文件末尾反向读取（tail 语义），`offset` 为"距末尾的行数"，offset=0 返回最新 N 行
- 行数上限：单次最多 1000 行

```python
LOG_DIR = Path(__file__).resolve().parent.parent / "logs"
LOG_FILES = {"api": LOG_DIR / "api.log", "worker": LOG_DIR / "worker.log"}


def _read_log_lines(source: Optional[str], level: Optional[str], keyword: Optional[str],
                    from_dt: Optional[str], to_dt: Optional[str],
                    limit: int, offset: int) -> tuple[list[dict], int]:
    """从磁盘日志文件读取并过滤，返回 (entries, total_matched)。

    注意：此函数使用同步文件 I/O 和全量内存加载。实现时需：
    - 用 asyncio.to_thread() 包装以非阻塞方式执行
    - 若日志文件过大（>10MB），改为从文件末尾逐行倒读，读到满足 limit+offset 即停止
    """
    import json

    files_to_read = []
    if source:
        if source in LOG_FILES:
            files_to_read.append(LOG_FILES[source])
    else:
        files_to_read = list(LOG_FILES.values())

    all_lines: list[dict] = []
    for fp in files_to_read:
        if not fp.exists():
            continue
        with open(fp, "r", encoding="utf-8") as f:
            for line in f:
                line = line.strip()
                if not line:
                    continue
                try:
                    entry = json.loads(line)
                except json.JSONDecodeError:
                    continue
                all_lines.append(entry)

    # 时间范围过滤
    if from_dt:
        all_lines = [e for e in all_lines if e.get("timestamp", "") >= from_dt]
    if to_dt:
        all_lines = [e for e in all_lines if e.get("timestamp", "") <= to_dt]

    # 级别/关键字过滤
    if level:
        all_lines = [e for e in all_lines if e.get("level", "").upper() == level.upper()]
    if keyword:
        kw = keyword.lower()
        all_lines = [e for e in all_lines if kw in e.get("message", "").lower()]

    # 按时间戳倒序
    all_lines.sort(key=lambda e: e.get("timestamp", ""), reverse=True)

    total = len(all_lines)
    entries = all_lines[offset:offset + limit]
    return entries, total


@router.get("/v1/admin/logs", response_model=LogQueryResponse)
async def query_logs(
    level: Optional[str] = Query(default=None),
    source: Optional[str] = Query(default=None),
    keyword: Optional[str] = Query(default=None),
    from_dt: Optional[str] = Query(default=None, alias="from", description="ISO 8601 起始时间"),
    to_dt: Optional[str] = Query(default=None, alias="to", description="ISO 8601 结束时间"),
    limit: int = Query(default=100, ge=1, le=1000),
    offset: int = Query(default=0, ge=0),
):
    # asyncio.to_thread 避免同步文件 I/O 阻塞事件循环
    entries, total = await asyncio.to_thread(
        _read_log_lines, source, level, keyword, from_dt, to_dt, limit, offset,
    )
    return LogQueryResponse(
        entries=[LogEntry(**e) for e in entries],
        total=total,
        has_more=(offset + limit) < total,
    )
```

### 6.1 `app/logger.py`

```python
import json
import sys
from datetime import datetime, timezone
from pathlib import Path
from typing import Optional

LOG_DIR = Path(__file__).resolve().parent.parent / "logs"
LOG_FILE = LOG_DIR / "api.log"

class JsonFormatter(logging.Formatter):
    """JSON 行格式化器——与 Worker 完全一致。"""
    def format(self, record):
        entry = {
            "timestamp": datetime.now(tz=timezone.utc).isoformat(),
            "level": record.levelname,
            "logger": record.name,
            "message": record.getMessage(),
        }
        if hasattr(record, "extra_data"):
            entry["extra"] = record.extra_data
        return json.dumps(entry, ensure_ascii=False)


def get_logger(name: str = "api") -> logging.Logger:
    global _logger
    if _logger is not None:
        return _logger

    LOG_DIR.mkdir(parents=True, exist_ok=True)

    # 文件输出：JSON 格式，按天轮转，保留 7 天
    from logging.handlers import TimedRotatingFileHandler
    json_fmt = JsonFormatter()
    fh = TimedRotatingFileHandler(str(LOG_FILE), when="D", interval=1, backupCount=7, encoding="utf-8")
    fh.setLevel(logging.DEBUG)
    fh.setFormatter(json_fmt)

    # 控制台输出：可读格式，INFO 及以上
    console_fmt = logging.Formatter(
        "%(asctime)s [%(levelname)-5s] %(name)s %(message)s",
        datefmt="%Y-%m-%d %H:%M:%S",
    )
    ch = logging.StreamHandler(sys.stderr)
    ch.setLevel(logging.INFO)
    ch.setFormatter(console_fmt)

    _logger = logging.getLogger(name)
    _logger.setLevel(logging.DEBUG)
    _logger.addHandler(fh)
    _logger.addHandler(ch)

    return _logger
```

## 9. 技术约束

### 9.1 部署顺序

1. 执行 `db/migrations/v0.2.sql`（重命名列 + 新增列 + 移除约束）
2. 执行数据清理脚本（3.2.3 节 SQL）
3. 部署新代码（API + Worker + 前端）
4. 前端首次加载处理空状态（引导提示）

### 9.2 API 导入 Worker 模块

API 层 `POST /v1/sources/{id}/verify` 导入 `worker/fetch_x_twitter.parse_x_twitter`。该函数内部创建自己的 `httpx.AsyncClient`，不依赖 Worker 数据库连接，交叉导入是安全的。代码路径：`app/routes.py` → `worker/fetch_x_twitter.py`（纯抓取逻辑）→ 返回结果。

### 9.3 日志格式统一

API 和 Worker 的文件日志均使用 JSON 行格式（通过 `JsonFormatter`）。控制台输出保持人类可读格式。`GET /v1/admin/logs` 依赖 JSON 行格式做反序列化和过滤。若日志文件包含非 JSON 行（如启动横幅），读取时静默跳过。

### 9.4 日志轮转

API 和 Worker 的文件日志均使用 `logging.handlers.TimedRotatingFileHandler`（Python 标准库），配置 `when="D"`（按天轮转）、`backupCount=7`（保留 7 天），满足 PRD 3.4.1 要求。轮转由 handler 内置支持，无需外部 cron/logrotate。

### 9.5 Source 删除性能

DELETE Source 的级联范围覆盖 5 张表（channel_sources → tasks/raw_items/source_states + raw_items → processed_news）。对于一个活跃 Source（关联数百条 raw_items 和 processed_news），级联删除需要数秒。当前项目体量下（单用户、少量 Source）不会成为瓶颈，但前端应展示 loading 状态。

a) **同步阻塞**：函数使用 `open(fp)` 同步读取文件，在 async 路由中直接调用会阻塞事件循环。实现时需用 `asyncio.to_thread()` 或 `loop.run_in_executor()` 包装。

**#2 `source_detector.py` 域名匹配存在误判——含路径的规则永远不生效（4.1）**

`detect_type_from_url` 中的匹配逻辑：

```python
for pattern, st in _DOMAIN_RULES:
    if pattern in host:  # host 仅为主机名，不含路径
        return st
```

`host` 来自 `urlparse(source_url).hostname`，仅包含域名部分（如 `huggingface.co`）。但规则 `"huggingface.co/papers"` 包含了路径 `/papers`，因此 `"huggingface.co/papers" in "huggingface.co"` 永远为 `False`。这意味着 **HuggingFace Daily Papers 来源无法被自动识别**。

此外，`if pattern in host` 的纯子串匹配存在误判风险：
- `"x.com" in "x.com.evil.com"` → True（误匹配，evil.com 的子域名碰巧含 x.com）
- `"github.com" in "notgithub.com"` → True（边界 case，实际较罕见）

**建议修正**：

```python
_DOMAIN_RULES: list[tuple[str, str, SourceType]] = [
    # (host_suffix, path_prefix, type)
    ("x.com", "", SourceType.x_twitter),
    ("twitter.com", "", SourceType.x_twitter),
    ("github.com", "", SourceType.github_trending),
    ("huggingface.co", "/papers", SourceType.hf_daily_papers),
    ("news.ycombinator.com", "", SourceType.hacker_news),
    ("semanticscholar.org", "", SourceType.semantic_scholar),
]
```

匹配逻辑改为精确域名后缀匹配 + 可选路径前缀匹配：

```python
def _host_matches(host: str, suffix: str) -> bool:
    return host == suffix or host.endswith("." + suffix)

for host_suffix, path_prefix, st in _DOMAIN_RULES:
    if _host_matches(host, host_suffix):
        if not path_prefix or path.startswith(path_prefix):
            return st
```
